    function formatNumber(value) {
  let formattedValue = parseFloat(value).toFixed(2); // Fix to two decimal places
  return formattedValue.endsWith('.00') ? parseInt(value) : formattedValue; // Remove decimal part if it's .00
}

function changeDateFormat(inputDate) {

    if(inputDate == ""){
        return "";
    }
    // Split the date string into year, month, and day
    var parts = inputDate.split('-');
    
    // Rearrange the parts to form the new date string in dd-mm-yyyy format
    var newDate = parts[2] + '-' + parts[1] + '-' + parts[0];
    
    return newDate;
}

    function sortTableByDate() {
        var table = document.getElementById("combinedTable");
        if (table) {
            var rows = Array.from(table.rows).slice(1); // Exclude the header row
            var dateFormat = /[0-9]{2}-[0-9]{2}-[0-9]{4}/; // Regular expression to match the date format dd-mm-yyyy

            // Sort the rows based on the date values
            rows.sort(function(rowA, rowB) {
                var dateA = rowA.cells[1].innerText.match(dateFormat)[0]; // Extract the date string from the second cell
                var dateB = rowB.cells[1].innerText.match(dateFormat)[0];

                // Convert the date strings to a format that can be compared (e.g., yyyy-mm-dd)
                var datePartsA = dateA.split('-');
                var datePartsB = dateB.split('-');
                var dateComparableA = datePartsA[2] + '-' + datePartsA[1] + '-' + datePartsA[0];
                var dateComparableB = datePartsB[2] + '-' + datePartsB[1] + '-' + datePartsB[0];

                return new Date(dateComparableA) - new Date(dateComparableB);
            });

            // Reorder the rows in the table
            rows.forEach(function(row) {
                table.appendChild(row);
            });
        } else {
            console.error("Table with ID 'combinedTable' not found.");
        }
    }
